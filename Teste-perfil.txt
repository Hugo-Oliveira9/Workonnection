  <script>
  document.addEventListener("DOMContentLoaded", () => {
    // Modal Editar Perfil
    const modalPerfil = new bootstrap.Modal(document.getElementById("editarPerfilModal"));
    document.getElementById("btnAbrirModal").addEventListener("click", () => modalPerfil.show());
    document.getElementById("btnSalvarModal").addEventListener("click", () => {
      document.getElementById("perfil-nome").textContent = "Nome: " + document.getElementById("inputNome").value;
      document.getElementById("perfil-formacao").textContent = "Email: " + document.getElementById("inputEmail").value;
      document.getElementById("perfil-local").textContent = "Local: " + document.getElementById("inputLocal").value;
      document.getElementById("perfil-telefone").textContent = "Telefone: " + document.getElementById("inputTelefone").value;
      document.getElementById("perfil-instagram").textContent = "Instagram: " + document.getElementById("inputInstagram").value;
      document.getElementById("perfil-linkedin").textContent = "LinkedIn: " + document.getElementById("inputLinkedin").value;
      document.getElementById("perfil-site").textContent = "Site: " + document.getElementById("inputSite").value;
      const foto = document.getElementById("inputFoto").value;
      if(foto) document.getElementById("perfil-foto").src = foto;
      modalPerfil.hide();
    });

    // Modal Sobre
    const modalSobre = new bootstrap.Modal(document.getElementById("editarSobreModal"));
    document.getElementById("btnEditarSobre").addEventListener("click", () => {
      document.getElementById("inputSobre").value = document.getElementById("sobre-conteudo").textContent;
      modalSobre.show();
    });
    document.getElementById("btnSalvarSobre").addEventListener("click", () => {
      document.getElementById("sobre-conteudo").textContent = document.getElementById("inputSobre").value;
      modalSobre.hide();
    });

    // Modal Habilidades
    const modalHabilidades = new bootstrap.Modal(document.getElementById("modalHabilidades"));
    document.getElementById("btnEditarHabilidades").addEventListener("click", () => modalHabilidades.show());
    document.getElementById("btnSalvarHabilidades").addEventListener("click", () => {
      const hab = document.getElementById("inputHabilidade").value;
      if(hab) {
        const span = document.createElement("span");
        span.className = "badge bg-primary me-1 mb-1";
        span.textContent = hab;
        document.getElementById("habilidades-lista").appendChild(span);
        document.getElementById("inputHabilidade").value = "";
      }
      modalHabilidades.hide();
    });

    // Modal Formação
    const modalFormacao = new bootstrap.Modal(document.getElementById("modalFormacao"));
    document.getElementById("btnEditarFormacao").addEventListener("click", () => modalFormacao.show());
    document.getElementById("btnSalvarFormacao").addEventListener("click", () => {
      const uni = document.getElementById("inputUniversidade").value;
      const curso = document.getElementById("inputCurso").value;
      const periodo = document.getElementById("inputPeriodo").value;
      if(uni && curso && periodo) {
        const div = document.createElement("div");
        div.className = "formacao-item";
        div.innerHTML = `<i class="fas fa-university text-primary"></i><div><b>${curso}</b><br><span>${uni}</span><br><span>${periodo}</span></div>`;
        document.getElementById("formacoes-lista").appendChild(div);
        document.getElementById("inputUniversidade").value = "";
        document.getElementById("inputCurso").value = "";
        document.getElementById("inputPeriodo").value = "";
      }
      modalFormacao.hide();
    });

    // Modal Experiências
    const modalExp = new bootstrap.Modal(document.getElementById("modalExperiencias"));
    document.getElementById("btnEditarExperiencias").addEventListener("click", () => modalExp.show());
    document.getElementById("btnSalvarExperiencia").addEventListener("click", () => {
      const empresa = document.getElementById("inputEmpresa").value;
      const cargo = document.getElementById("inputCargo").value;
      const desc = document.getElementById("inputDescricao").value;
      const local = document.getElementById("inputLocalExp").value;
      const periodo = document.getElementById("inputPeriodoExp").value;
      if(empresa && cargo && desc && local && periodo) {
        const div = document.createElement("div");
        div.className = "experiencia-item";
        div.innerHTML = `<i class="fas fa-building text-primary"></i><div><b>${cargo}</b><br><span>${empresa} - ${local}</span><br><span>${desc}</span><br><span>${periodo}</span></div>`;
        document.getElementById("experiencias-lista").appendChild(div);
        document.getElementById("inputEmpresa").value = "";
        document.getElementById("inputCargo").value = "";
        document.getElementById("inputDescricao").value = "";
        document.getElementById("inputLocalExp").value = "";
        document.getElementById("inputPeriodoExp").value = "";
      }
      modalExp.hide();
    });

    // Modal Cursos
    const modalCursos = new bootstrap.Modal(document.getElementById("modalCursos"));
    document.getElementById("btnEditarCursos")?.addEventListener("click", () => modalCursos.show());
    document.getElementById("btnSalvarCurso")?.addEventListener("click", () => {
      const nome = document.getElementById("inputNomeCurso").value;
      const inst = document.getElementById("inputInstituicaoCurso").value;
      const periodo = document.getElementById("inputPeriodoCurso").value;
      if(nome && inst && periodo) {
        const div = document.createElement("div");
        div.className = "experiencia-item";
        div.innerHTML = `<i class="fas fa-book text-primary"></i><div><b>${nome}</b><br><span>${inst}</span><br><span>${periodo}</span></div>`;
        document.getElementById("cursos-lista").appendChild(div);
        document.getElementById("inputNomeCurso").value = "";
        document.getElementById("inputInstituicaoCurso").value = "";
        document.getElementById("inputPeriodoCurso").value = "";
      }
      modalCursos.hide();
    });
  });
  </script>
